223:def compute_rsi(closes, period=14):
224:    if not closes or len(closes) < period + 1:
225:        return None
226:    try:
227:        import pandas as pd
228:        s = pd.Series(closes)
229:        delta = s.diff()
230:        gain = delta.clip(lower=0)
231:        loss = -delta.clip(upper=0)
232:        avg_gain = gain.ewm(alpha=1/period, min_periods=period, adjust=False).mean()
233:        avg_loss = loss.ewm(alpha=1/period, min_periods=period, adjust=False).mean()
234:        rs = avg_gain / avg_loss
235:        rsi = 100 - (100 / (1 + rs))
236:        result = round(float(rsi.iloc[-1]), 1)
237:        return max(1.0, min(99.0, result))
238:    except Exception as e:
239:        print(f"  RSI calculation error: {e}")
240:        return None

249:# ── SANITIZE: replace float NaN/Inf with None ─────────────────────────────────
251:def sanitize(obj):
252:    """
253:    Recursively walk the data structure and replace any float NaN or Infinity
254:    with None so json.dump() never writes invalid JSON tokens.
255:    This is the permanent fix — NaN values from pandas/yfinance calculations
256:    are caught here before they ever reach cache.json.
257:    """
258:    if isinstance(obj, dict):
259:        return {k: sanitize(v) for k, v in obj.items()}
260:    elif isinstance(obj, list):
261:        return [sanitize(v) for v in obj]
262:    elif isinstance(obj, float):
263:        if math.isnan(obj) or math.isinf(obj):
264:            return None
265:        return obj
266:    return obj
